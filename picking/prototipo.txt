  <script>
    const defaultConfig = {
      background_color: "#0f172a",
      surface_color: "#ffffff",
      text_color: "#1e293b",
      primary_action_color: "#10b981",
      secondary_action_color: "#f59e0b",
      page_title: "Registro de Ocorr√™ncias",
      new_record_button: "Inserir Nova Linha",
      settings_button: "Ajustar Listas de Gabarito",
      search_placeholder: "Buscar ocorr√™ncias...",
      font_family: "Inter",
      font_size: 14
    };
    
    let allRecords = [];
    let editingCell = null;
    
    const dataHandler = {
      onDataChanged(data) {
        allRecords = data;
        renderTable(data);
        updateEmptyState(data.length === 0);
      }
    };
    
    function updateEmptyState(isEmpty) {
      const emptyState = document.getElementById('empty-state');
      const tableContainer = document.querySelector('.table-scrollable');
      
      if (isEmpty) {
        emptyState.style.display = 'block';
        tableContainer.style.display = 'none';
      } else {
        emptyState.style.display = 'none';
        tableContainer.style.display = 'block';
      }
    }
    
    function renderTable(records) {
      const tbody = document.getElementById('table-body');
      tbody.innerHTML = '';
      
      records.forEach(record => {
        const row = createTableRow(record);
        tbody.appendChild(row);
      });
    }
    
    function createTableRow(record) {
      const row = document.createElement('tr');
      row.className = 'border-b hover:bg-opacity-50 transition-colors';
      row.dataset.recordId = record.__backendId;
      
      const fields = ['data', 'hora', 'tipo', 'prioridade', 'responsavel', 'status', 'local', 'descricao', 'observacoes', 'protocolo', 'categoria', 'subcategoria', 'departamento', 'solicitante', 'prazo'];
      
      fields.forEach(field => {
        const cell = document.createElement('td');
        cell.className = 'px-4 py-3 text-sm border-r table-cell-editable whitespace-nowrap';
        cell.textContent = record[field] || '';
        cell.dataset.field = field;
        cell.addEventListener('click', () => startEditing(cell, record));
        row.appendChild(cell);
      });
      
      const actionsCell = document.createElement('td');
      actionsCell.className = 'px-4 py-3 text-sm';
      const deleteBtn = document.createElement('button');
      deleteBtn.textContent = 'üóëÔ∏è';
      deleteBtn.className = 'hover:scale-125 transition-transform';
      deleteBtn.addEventListener('click', () => deleteRecord(record));
      actionsCell.appendChild(deleteBtn);
      row.appendChild(actionsCell);
      
      return row;
    }
    
    function startEditing(cell, record) {
      if (editingCell) {
        finishEditing();
      }
      
      editingCell = { cell, record, originalValue: cell.textContent };
      cell.classList.add('table-cell-editing');
      
      const input = document.createElement('input');
      input.type = 'text';
      input.className = 'cell-input';
      input.value = cell.textContent;
      
      cell.textContent = '';
      cell.appendChild(input);
      input.focus();
      input.select();
      
      input.addEventListener('blur', finishEditing);
      input.addEventListener('keydown', (e) => {
        if (e.key === 'Enter') {
          finishEditing();
        } else if (e.key === 'Escape') {
          cancelEditing();
        }
      });
    }
    
    async function finishEditing() {
      if (!editingCell) return;
      
      const { cell, record } = editingCell;
      const input = cell.querySelector('input');
      const newValue = input ? input.value : cell.textContent;
      const field = cell.dataset.field;
      
      cell.classList.remove('table-cell-editing');
      cell.textContent = newValue;
      
      if (newValue !== editingCell.originalValue) {
        const updatedRecord = { ...record, [field]: newValue };
        const result = await window.dataSdk.update(updatedRecord);
        
        if (!result.isOk) {
          cell.textContent = editingCell.originalValue;
          showToast('Erro ao atualizar: ' + result.error.message, 'error');
        }
      }
      
      editingCell = null;
    }
    
    function cancelEditing() {
      if (!editingCell) return;
      
      const { cell } = editingCell;
      cell.classList.remove('table-cell-editing');
      cell.textContent = editingCell.originalValue;
      editingCell = null;
    }
    
    async function deleteRecord(record) {
      const deleteBtn = event.target;
      const originalText = deleteBtn.textContent;
      
      if (deleteBtn.textContent !== '‚úÖ Confirmar?') {
        deleteBtn.textContent = '‚úÖ Confirmar?';
        setTimeout(() => {
          if (deleteBtn.textContent === '‚úÖ Confirmar?') {
            deleteBtn.textContent = originalText;
          }
        }, 3000);
        return;
      }
      
      deleteBtn.disabled = true;
      const result = await window.dataSdk.delete(record);
      
      if (!result.isOk) {
        showToast('Erro ao excluir: ' + result.error.message, 'error');
        deleteBtn.disabled = false;
        deleteBtn.textContent = originalText;
      }
    }
    
    function showToast(message, type = 'info') {
      const toast = document.createElement('div');
      toast.className = 'fixed top-4 right-4 px-6 py-3 rounded-lg shadow-lg text-white z-50 transition-opacity';
      toast.style.backgroundColor = type === 'error' ? '#ef4444' : '#10b981';
      toast.textContent = message;
      
      document.body.appendChild(toast);
      
      setTimeout(() => {
        toast.style.opacity = '0';
        setTimeout(() => toast.remove(), 300);
      }, 3000);
    }
    
    async function insertNewRow() {
      if (allRecords.length >= 999) {
        showToast('Limite m√°ximo de 999 registros atingido. Exclua alguns registros primeiro.', 'error');
        return;
      }
      
      const btnInsert = document.getElementById('btn-insert');
      btnInsert.disabled = true;
      btnInsert.style.opacity = '0.5';
      
      const newRecord = {
        id: Date.now().toString(),
        data: new Date().toLocaleDateString('pt-BR'),
        hora: new Date().toLocaleTimeString('pt-BR', { hour: '2-digit', minute: '2-digit' }),
        tipo: '',
        prioridade: '',
        responsavel: '',
        status: 'Novo',
        local: '',
        descricao: '',
        observacoes: '',
        protocolo: 'PROT-' + Date.now(),
        categoria: '',
        subcategoria: '',
        departamento: '',
        solicitante: '',
        prazo: ''
      };
      
      const result = await window.dataSdk.create(newRecord);
      
      btnInsert.disabled = false;
      btnInsert.style.opacity = '1';
      
      if (!result.isOk) {
        showToast('Erro ao criar registro: ' + result.error.message, 'error');
      } else {
        showToast('Nova linha inserida com sucesso!', 'info');
      }
    }
    
    function setupSearch() {
      const searchInput = document.getElementById('search-input');
      searchInput.addEventListener('input', (e) => {
        const searchTerm = e.target.value.toLowerCase();
        
        if (!searchTerm) {
          renderTable(allRecords);
          return;
        }
        
        const filtered = allRecords.filter(record => {
          return Object.values(record).some(value => 
            String(value).toLowerCase().includes(searchTerm)
          );
        });
        
        renderTable(filtered);
      });
    }
